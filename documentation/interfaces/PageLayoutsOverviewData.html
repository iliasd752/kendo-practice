import { Component, OnInit, ViewEncapsulation } from '@angular/core';
import { ActivatedRoute } from '@angular/router';

export interface PageLayoutsOverviewData
{
    title: string;
    description: string;
    availableOptions: {
        value: string;
        title: string;
    }[];
    selectedOption: string;
    options: {
        [option: string]: {
            description: string;
            link: string;
            component: any;
        };
    };
}
